//http.js

import axios from "axios";

// 创建axios实例
const httpInstance = axios.create({
  // 后端由黑马提供
  baseURL: "http://pcapi-xiaotuxian-front-devtest.itheima.net",
  timeout: 5000,
});

/*
请求和响应拦截器可以用来在请求发送到服务器之前或响应返回给客户端之前，对请求和响应进行统一的预处理操作
*/

// axios请求拦截器
httpInstance.interceptors.request.use(
  (config) => {
    return config;
  },
  (e) => Promise.reject(e),
);

// axios响应式拦截器
httpInstance.interceptors.response.use(
  (res) => res.data,
  (e) => {
    return Promise.reject(e);
  },
);

export default httpInstance;

该js文件通过export暴露出实例httpInstance，以后对不同路由的接口封装可以引用这个实例封装API函数接口

拦截器

这里定义了axios的请求拦截器和响应拦截器

在使用 axios 时，请求和响应拦截器可以用来在请求发送到服务器之前或响应返回给客户端之前，对请求和响应进行统一的预处理操作。以下是它们的主要作用：

请求拦截器的作用

  1. 统一处理请求配置 在每个请求发送前修改请求头、请求参数等，例如添加认证 token、设置全局的 Content-Type。

    axios.interceptors.request.use(
    config => {
        // 在发送请求前做些什么
        config.headers.Authorization = `Bearer ${token}`;
        return config;
    },
    error => {
        // 对请求错误做些什么
        return Promise.reject(error);
    }
);

  2. 添加全局 loading 动画 可以在请求发送前开启加载动画，确保用户知道有操作正在进行。

    1
2
3
4

    axios.interceptors.request.use(config => {
    showLoading(); // 开启loading动画
    return config;
});

  3. 参数校验和规范化 在请求发送前对参数进行校验，或者将其格式化为服务器所需的结构。

响应拦截器的作用

  1. 统一处理响应数据 对所有响应进行统一处理，例如只返回业务需要的部分数据，剥离掉不必要的信息。

    1
2

    axios.interceptors.response.use(
    response => {
        // 只返回实际业务数据部分
        return response.data;
    },
    error => {
        return Promise.reject(error);
    }
);

  2. 全局错误处理 拦截服务器返回的错误（如 HTTP 状态码 4xx 或 5xx），进行统一处理，如弹出错误提示、跳转到登录页等。

    1

    axios.interceptors.response.use(
    response => response,
    error => {
        if (error.response.status === 401) {
            redirectToLogin(); // 未授权，跳转登录
        }
        return Promise.reject(error);
    }
);

  3. 自动刷新 Token 如果服务器返回 Token 过期错误，可以在响应拦截器中拦截此错误，发起刷新 Token 的请求后重新尝试原来的请求。

  4. 关闭全局 loading 动画 请求完成后关闭 loading 动画。

        hideLoading();

    }
);

总结

  • 请求拦截器：主要用于在请求发出前进行 token 注入、参数校验、设置全局配置等。
  • 响应拦截器：主要用于统一处理响应数据、错误处理、业务逻辑预处理等。

这种机制极大地方便了全局管理 HTTP 请求和响应逻辑。
