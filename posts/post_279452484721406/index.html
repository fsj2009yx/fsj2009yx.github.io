int a[100001];
int b[100001] = {0};
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    int n;
    cin >> n;
    ll ans = 0;
    for (int i = 1; i <= n; i++)
    {
        cin >> a[i];
        b[i] = b[i - 1] ^ a[i];
        ans += b[i];
    }
    for (int i = n; i >= 1; i--)
    {
        for (int j = 1; j < i; j++)
        {
            if (b[i] == b[j])
            {
                continue;
            }
            else
            {
                ans += (b[i] ^ b[j]);
            }
        }
    }
    cout << ans << endl;
}
